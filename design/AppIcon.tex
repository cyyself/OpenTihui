\newcommand{\sparkle}[3]{%
  \begin{scope}[shift={(#1,#2)}]
    \fill[white] (0,#3)
      .. controls ({0.16*#3},{0.16*#3}) .. (#3,0)
      .. controls ({0.16*#3},{-0.16*#3}) .. (0,-#3)
      .. controls ({-0.16*#3},{-0.16*#3}) .. (-#3,0)
      .. controls ({-0.16*#3},{0.16*#3}) .. (0,#3) -- cycle;
  \end{scope}}
